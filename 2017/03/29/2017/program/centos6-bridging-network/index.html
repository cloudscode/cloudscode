Centos6 桥接网络配置

编辑配置文件

添加修改以下内容

DEVICE=eth0
HWADDR=08:00:27:BC:56:DB
TYPE=Ethernet
UUID=8247fb0d-ca65-4c4b-ab61-c21e1c8f5406
ONBOOT=yes #开启自动启用网络连接
NM_CONTROLLED=yes
BOOTPROTO=static #启用静态IP地址
IPADDR0=192.168.3.236 #设置IP地址
NETMASK=255.255.255.0
GATEWAY0=192.168.3.1 #设置网关
PREFIXO0=24 #设置子网掩码
DNS1=218.85.157.99 #设置主DNS
DNS2=218.85.157.99 #设置备DNS
PEERDNS=yes
USERCTL=no
IPV6INIT=no
PREFIX=24
DEFROUTE=yes

保存退出

重启网络

# centos
